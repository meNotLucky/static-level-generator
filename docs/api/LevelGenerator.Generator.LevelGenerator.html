    Examples

    This sample shows how to call the GenerateNewLevel() method from an ExampleClass component on a GameObject that also has a LevelGenerator component.

    using LevelGenerator.Generator;

class ExampleClass : MonoBehaviour
{
    private void Start()
    {
        LevelGenerator.GenerateNewLevel();
    }
}
